if / else if / else

if (x > 0) {
  console.log('додатне');
} else if (x === 0) {
  console.log('нуль');
} else {
  console.log('від'ємне');
}

Тернарний оператор

let res = x > 0 ? 'плюс' : 'мінус';

switch

switch (day) {
  case 1: console.log('Пн'); break;
  case 2: console.log('Вт'); break;
  default: console.log('інший');
}

Логічні оператори

&&  — і (обидві умови true)
||  — або (хоча б одна true)
!   — заперечення

Цикли

// for — відома кількість кроків
for (let i = 0; i < 5; i++) { ... }

// while — поки умова true
while (i < 5) { i++; }

// do-while — виконується хоча б раз
do { i++; } while (i < 5);

// for-in — ключі об'єкта
for (let key in obj) { ... }

// for-of — значення масиву
for (let val of arr) { ... }

break та continue

for (let i = 0; i < 10; i++) {
  if (i === 3) continue; // пропустити i=3
  if (i === 7) break;    // вийти при i=7
  console.log(i);
}
