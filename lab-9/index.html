Змінні

var x = 1;   // функціональна область
let y = 2;   // блокова область
const Z = 3; // константа, не перезаписується

Типи даних

number    → 42, 3.14
string    → 'текст', "текст"
boolean   → true, false
null      → явна відсутність значення
undefined → змінна оголошена, але без значення
object    → {key: val}, [], null
symbol, bigint → рідко вживані

Діалогові вікна

alert('Повідомлення');
let s = prompt('Питання', 'за замовч.');
let ok = confirm('Певен?'); // true / false

Перетворення типів

parseInt('42px')   // → 42
parseFloat('3.14') // → 3.14
Number('42')       // → 42
String(42)         // → '42'
Boolean(0)         // → false
Boolean('a')       // → true

Об'єкт Math

Math.round(4.6)   // 5
Math.floor(4.9)   // 4
Math.ceil(4.1)    // 5
Math.abs(-7)      // 7
Math.max(1,5,3)   // 5
Math.min(1,5,3)   // 1
Math.sqrt(9)      // 3
Math.pow(2, 8)    // 256
Math.PI           // 3.14159...
Math.random()     // 0..1 (не включно з 1)
