Способи оголошення

// Function Declaration (hoisting є)
function add(a, b) { return a + b; }

// Function Expression (hoisting немає)
const add = function(a, b) { return a + b; };

// Стрілкова функція
const add = (a, b) => a + b;
const square = x => x * x; // один параметр

Параметри за замовчуванням

function greet(name = 'Гість') {
  return `Привіт, ${name}!`;
}
greet();         // 'Привіт, Гість!'
greet('Богдан'); // 'Привіт, Богдан!'

Об'єкт arguments

function sum() {
  let total = 0;
  for (let v of arguments) total += v;
  return total;
}
sum(1, 2, 3, 4); // 10

Callback (функція як аргумент)

function run(fn) { fn(); }
run(() => console.log('виконано'));

// Класичний приклад
[1,2,3].forEach(function(x) {
  console.log(x);
});

Об'єкт Date

const d = new Date();
d.getFullYear()  // рік (2026)
d.getMonth()     // місяць 0–11 (!)
d.getDate()      // день місяця 1–31
d.getDay()       // день тижня 0–6, 0=нд
d.getHours()     // години
d.getTime()      // мс від 01.01.1970

// Різниця в днях
const diff = (new Date('2026-12-31') - new Date()) / 86400000;
